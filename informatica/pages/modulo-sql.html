SQL
il linguaggio dei dati

Dal DDL per creare strutture, al DML per manipolare dati, fino a JOIN, subquery e oggetti avanzati.

DDL — Data Definition Language

  • CREATE TABLE · ALTER TABLE · DROP TABLE · TRUNCATE TABLE
  • Tipi di dato: INT, VARCHAR, TEXT, DATE, DATETIME, DECIMAL, BOOLEAN
  • Vincoli: NOT NULL, UNIQUE, PRIMARY KEY, FOREIGN KEY, CHECK, DEFAULT, AUTO_INCREMENT

DML — Data Manipulation Language

  • INSERT INTO · UPDATE ... SET · DELETE FROM
  • SELECT con WHERE, ORDER BY, GROUP BY, HAVING, LIMIT, DISTINCT
  • Funzioni: COUNT, SUM, AVG, MAX, MIN · DATE_FORMAT(), IF(), CONCAT()

Query avanzate

  • INNER JOIN, LEFT JOIN — query multi-tabella
  • Subquery, ANY, ALL, IN, BETWEEN, BETWEEN
  • UNION vs UNION ALL

Oggetti SQL

  • VIEW — tabella virtuale
  • Trigger — procedura automatica su eventi DML
  • Stored procedure — blocco SQL riutilizzabile
  • Indici — ottimizzazione delle ricerche
«SQL è il linguaggio con cui parli al database. Impararlo bene significa non aver mai paura dei dati.» — Sviluppo web
Sezione 02

DDL — Definizione della struttura

CREATE TABLE Studenti (
    id        INT          AUTO_INCREMENT PRIMARY KEY,
    matricola VARCHAR(10)  NOT NULL UNIQUE,
    nome      VARCHAR(50)  NOT NULL,
    cognome   VARCHAR(50)  NOT NULL,
    email     VARCHAR(100) UNIQUE,
    voto      INT          CHECK (voto BETWEEN 18 AND 30),
    data_iscr DATE         DEFAULT (CURDATE()),
    id_corso  INT,
    FOREIGN KEY (id_corso) REFERENCES Corsi(id)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);

-- Aggiungi colonna
ALTER TABLE Studenti ADD COLUMN telefono VARCHAR(20);

-- Modifica tipo colonna
ALTER TABLE Studenti MODIFY COLUMN voto DECIMAL(4,2);

-- Aggiungi vincolo
ALTER TABLE Studenti ADD CONSTRAINT chk_voto CHECK (voto >= 18);

-- Aggiungi indice
ALTER TABLE Studenti ADD INDEX idx_cognome (cognome);

-- Rinomina tabella
ALTER TABLE Studenti RENAME TO Iscritti;

-- Elimina tabella (struttura + dati)
DROP TABLE IF EXISTS Studenti;

-- Svuota tabella (mantiene struttura, azzera AUTO_INCREMENT)
TRUNCATE TABLE Studenti;

TRUNCATE vs DELETE FROM

TRUNCATE: svuota la tabella in un'unica operazione, più veloce, azzera AUTO_INCREMENT, non loggabile riga per riga, non attiva trigger. DELETE FROM senza WHERE: elimina tutte le righe una per una, loggabile, attiva trigger, NON azzera AUTO_INCREMENT. Se usi una FK che referenzia la tabella, TRUNCATE potrebbe fallire.

Sezione 03

SELECT avanzato

-- Struttura completa (ordine OBBLIGATORIO)
SELECT [DISTINCT] colonne
FROM tabella
[WHERE condizione]
[GROUP BY colonne]
[HAVING condizione_gruppo]
[ORDER BY colonne [ASC|DESC]]
[LIMIT n [OFFSET m]];

-- Esempio completo
SELECT categoria, COUNT(*) AS totale, AVG(prezzo) AS prezzo_medio
FROM Prodotti
WHERE attivo = 1
GROUP BY categoria
HAVING totale > 5
ORDER BY prezzo_medio DESC
LIMIT 10;

WHERE vs HAVING — differenza fondamentale

WHERE filtra le righe singole PRIMA del raggruppamento — non può usare funzioni aggregate. HAVING filtra i gruppi DOPO GROUP BY — può usare COUNT, SUM, AVG ecc.

-- Aggregazione
SELECT COUNT(*) AS totale_righe FROM Ordini;
SELECT COUNT(email) AS con_email FROM Utenti; -- ignora NULL
SELECT SUM(importo) AS totale FROM Fatture;
SELECT AVG(voto) AS media FROM Studenti WHERE voto IS NOT NULL;
SELECT MAX(stipendio), MIN(stipendio) FROM Dipendenti;

-- Funzioni su stringhe
SELECT UPPER(nome), LOWER(cognome), LENGTH(email) FROM Utenti;
SELECT CONCAT(nome, ' ', cognome) AS nominativo FROM Studenti;
SELECT SUBSTRING(testo, 1, 100) AS estratto FROM Articoli;

-- Funzioni su date
SELECT DATE_FORMAT(data_nascita, '%d/%m/%Y') AS nata FROM Persone;
SELECT YEAR(data_ordine), MONTH(data_ordine) FROM Ordini;
SELECT DATEDIFF(NOW(), data_assunzione) AS giorni_azienda FROM Dipendenti;

-- IF / CASE
SELECT nome, IF(voto >= 60, 'Promosso', 'Bocciato') AS esito FROM Studenti;
SELECT nome,
    CASE WHEN voto >= 90 THEN 'Ottimo'
         WHEN voto >= 70 THEN 'Buono'
         ELSE 'Sufficiente' END AS livello
FROM Studenti;
Sezione 04

JOIN e Subquery

INNER JOINLEFT JOIN
RestituisceSolo righe con corrispondenza in ENTRAMBE le tabelleTUTTE le righe della tabella sinistra; NULL dove non c'è corrispondenza a destra
UsoQuando la relazione è sempre presente (es. ordini con cliente esistente)Quando voglio anche i record senza corrispondenza (es. clienti SENZA ordini)
 
-- INNER JOIN: ordini con dati cliente
SELECT o.id, c.nome, o.importo
FROM Ordini o
INNER JOIN Clienti c ON o.id_cliente = c.id;

-- LEFT JOIN: clienti CON O SENZA ordini
SELECT c.nome, COUNT(o.id) AS num_ordini
FROM Clienti c
LEFT JOIN Ordini o ON c.id = o.id_cliente
GROUP BY c.id, c.nome
ORDER BY num_ordini DESC;

-- JOIN multiplo (più tabelle)
SELECT s.nome, c.titolo, i.data_iscrizione
FROM Studenti s
JOIN Iscrizioni i ON s.id = i.id_studente
JOIN Corsi c ON i.id_corso = c.id
WHERE c.anno = 2026;
-- Subquery non correlata (eseguita una volta)
SELECT nome FROM Studenti
WHERE voto > (SELECT AVG(voto) FROM Studenti);

-- Subquery con IN
SELECT nome FROM Prodotti
WHERE id_categoria IN (SELECT id FROM Categorie WHERE attiva = 1);

-- Subquery con EXISTS
SELECT nome FROM Clienti c
WHERE EXISTS (SELECT 1 FROM Ordini o WHERE o.id_cliente = c.id);

-- Subquery correlata (eseguita per ogni riga esterna — più lenta)
SELECT s.nome, s.voto
FROM Studenti s
WHERE s.voto = (SELECT MAX(s2.voto) FROM Studenti s2 WHERE s2.classe = s.classe);

-- Operatori ANY e ALL
SELECT nome FROM Studenti
WHERE voto > ANY (SELECT voto FROM Studenti WHERE classe = 'A'); -- almeno uno
WHERE voto > ALL (SELECT voto FROM Studenti WHERE classe = 'A'); -- tutti
-- UNION: elimina duplicati (più lento)
SELECT nome, 'Studente' AS ruolo FROM Studenti
UNION
SELECT nome, 'Professore' AS ruolo FROM Professori;

-- UNION ALL: mantiene duplicati (più veloce)
SELECT id_prodotto FROM Ordini_2024
UNION ALL
SELECT id_prodotto FROM Ordini_2025;

Requisiti: stesso numero di colonne, tipi compatibili. Il nome delle colonne viene preso dalla prima SELECT.

Sezione 05

Viste, Trigger, Stored Procedure e Indici

-- Crea vista
CREATE VIEW v_studenti_promossi AS
    SELECT id, nome, cognome, voto
    FROM Studenti
    WHERE voto >= 60;

-- Usa vista come una normale tabella
SELECT * FROM v_studenti_promossi ORDER BY voto DESC;

-- Elimina vista
DROP VIEW IF EXISTS v_studenti_promossi;

Vantaggi: semplifica query complesse, nasconde la struttura interna, può limitare i dati visibili. Limitazioni: VIEW con GROUP BY o UNION non è aggiornabile (no INSERT/UPDATE).

DELIMITER $$
CREATE TRIGGER log_modifica_voto
AFTER UPDATE ON Studenti
FOR EACH ROW
BEGIN
    IF NEW.voto != OLD.voto THEN
        INSERT INTO Log_Voti (id_studente, voto_old, voto_new, data_mod)
        VALUES (OLD.id, OLD.voto, NEW.voto, NOW());
    END IF;
END$$
DELIMITER ;

Timing: BEFORE (prima dell'operazione) o AFTER (dopo). Evento: INSERT, UPDATE, DELETE. NEW = valori nuovi (dopo). OLD = valori vecchi (prima).

-- INSERT IGNORE: se c'è conflitto di chiave univoca, ignora silenziosamente
INSERT IGNORE INTO Categorie (id, nome) VALUES (5, 'Elettronica');

-- ON DUPLICATE KEY UPDATE: upsert — insert se non esiste, update se esiste
INSERT INTO Visite (ip, count) VALUES ('192.168.1.1', 1)
ON DUPLICATE KEY UPDATE count = count + 1;

-- Esempio con più colonne
INSERT INTO Prezzi (id_prod, data, prezzo) VALUES (10, CURDATE(), 29.99)
ON DUPLICATE KEY UPDATE prezzo = VALUES(prezzo);
